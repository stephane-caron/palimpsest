208 template <typename T>
209 const T &get(const std::string &key, const T &default_value) const {
210 auto it = map_.find(key);
211 if (it != map_.end()) {
212 if (it->second->is_map()) {
213 throw TypeError(__FILE__, __LINE__,
214 "Object at key \"" + key +
215 "\" is a dictionary, cannot get a single value "
216 "from it. Did you "
217 "mean to use operator()?");
218 }
219 try {
220 return it->second->value_.get_reference<T>();
221 } catch (const TypeError &e) {
222 throw TypeError(
223 __FILE__, __LINE__,
224 "Object for key \"" + key +
225 "\" does not have the same type as the stored type. Stored " +
226 it->second->value_.type_name() + " but requested " +
227 typeid(T).name() + ".");
228 }
229 }
230 return default_value;
231 }

248 template <typename T, typename... ArgsT, typename... Args>
249 T &insert(const std::string &key, Args &&...args) {
250 if (this->is_value()) {
251 throw TypeError(__FILE__, __LINE__,
252 "Cannot insert at key \"" + key +
253 "\" in non-dictionary object of type \"" +
254 value_.type_name() + "\".");
255 }
256 auto &child = this->operator()(key);
257 if (!child.is_empty()) {
258 spdlog::warn(
259 "[Dictionary::insert] Key \"{}\" already exists. Returning existing "
260 "value rather than creating a new one.",
261 key);
262 return get<T>(key);
263 }
264 child.value_.allocate<T>();
265 new (child.value_.buffer.get()) T(std::forward<Args>(args)...);
266 T &ret = child.value_.setup<T, ArgsT...>();
267 return ret;
268 }

396 template <typename T>
397 T pop(const std::string &key, const T &default_value) {
398 auto it = map_.find(key);
399 if (it == map_.end()) {
400 return default_value;
401 }
402 if (it->second->is_map()) {
403 throw TypeError(__FILE__, __LINE__,
404 "Object at key \"" + key +
405 "\" is a dictionary, cannot pop a single value "
406 "from it.");
407 }
408 try {
409 const T value = it->second->value_.get_reference<T>();
410 remove(key);
411 return value;
412 } catch (const TypeError &e) {
413 throw TypeError(
414 __FILE__, __LINE__,
415 "Object for key \"" + key +
416 "\" does not have the same type as the requested type. Stored " +
417 it->second->value_.type_name() + " but requested " +
418 typeid(T).name() + ".");
419 }
