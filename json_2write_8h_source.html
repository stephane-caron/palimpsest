232template <typename T>
233inline void write(std::ostream &stream, const std::vector<T> &vector) {
234 if (vector.empty()) {
235 return;
236 }
237 stream << "[";
238 for (const auto &it = vector.begin(); it != vector.end(); ++it) {
239 if (it != vector.begin()) {
240 stream << ", ";
241 }
242 stream << *it;
243 }
244 stream << "]";
245}
