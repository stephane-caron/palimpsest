274template <>
275inline void read(const mpack_node_t node, Eigen::Vector2d& value) {
276 if (mpack_node_type(node) != mpack_type_array) {
277 throw TypeError(
278 __FILE__, __LINE__,
279 std::string("Expecting an array, but deserialized node has type ") +
280 mpack_type_to_string(mpack_node_type(node)));
281 }
282 assert(mpack_node_array_length(node) == 2);
283 read<double>(mpack_node_array_at(node, 0), value.x());
284 read<double>(mpack_node_array_at(node, 1), value.y());
