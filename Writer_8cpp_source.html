45static void mpack_std_vector_writer_flush(mpack_writer_t *writer,
46 const char *data, size_t count) {
47 auto &buffer = *static_cast<std::vector<char> *>(writer->context);
48
49 if (data == writer->buffer) {
50 if (mpack_writer_buffer_used(writer) == count) {
51 // teardown, do nothing
52 return;
53 }
54 // otherwise leave the data in the buffer and just grow
55 writer->current = writer->buffer + count;
56 count = 0;
57 }
58
59 size_t used = mpack_writer_buffer_used(writer);
60 size_t size = mpack_writer_buffer_size(writer);
61
62 mpack_log("flush size %i used %i data %p buffer %p\n",
63 static_cast<int>(count), static_cast<int>(used), data,
64 writer->buffer);
65
66 mpack_assert(
67 data == writer->buffer || used + count > size,
68 "extra flush for %i but there is %i space left in the buffer! (%i/%i)",
69 static_cast<int>(count),
70 static_cast<int>(mpack_writer_buffer_left(writer)),
71 static_cast<int>(used), static_cast<int>(size));
72
73 // grow to fit the data
74 // TODO(gergondet): this really needs to correctly test for overflow
75 size_t new_size = size * 2;
76 while (new_size < used + count) {
77 new_size *= 2;
78 }
79
80 mpack_log("flush growing buffer size from %i to %i\n", static_cast<int>(size),
81 static_cast<int>(new_size));
82
83 // grow the buffer
84 buffer.resize(new_size);
85 char *new_buffer = buffer.data();
86 if (new_buffer == NULL) {
87 mpack_writer_flag_error(writer, mpack_error_memory);
88 return;
89 }
90 writer->current = new_buffer + used;
91 writer->buffer = new_buffer;
92 writer->end = writer->buffer + new_size;
93
94 // append the extra data
95 if (count > 0) {
96 mpack_memcpy(writer->current, data, count);
97 writer->current += count;
98 }
99
100 mpack_log("new buffer %p, used %i\n", new_buffer,
101 static_cast<int>(mpack_writer_buffer_used(writer)));
102}

106Writer::Writer(std::vector<char> &buffer) {
107 if (buffer.size() == 0) {
108 buffer.resize(MPACK_BUFFER_SIZE);
109 }
110 mpack_writer_init(&writer_, buffer.data(), buffer.size());
111 mpack_writer_set_context(&writer_, &buffer); // used in flush function
112 mpack_writer_set_flush(&writer_, mpack_std_vector_writer_flush);
113}

147template <typename T>
148inline void write_vector(mpack_writer_t *writer, const T &v) {
149 for (Eigen::Index i = 0; i < v.size(); ++i) {
150 mpack_write_double(writer, v(i));

154template <typename T>
155inline void write_matrix(mpack_writer_t *writer, const T &m) {
156 for (Eigen::Index i = 0; i < m.rows(); ++i) {
157 for (Eigen::Index j = 0; j < m.cols(); ++j) {
158 mpack_write_double(writer, m(i, j));
159 }
160 }
